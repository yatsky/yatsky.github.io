import urllib.request

def get_word_def(word: str) -> bytes:
    if len(word.split()) > 1:
        return "[]".encode()
    defn = urllib.request.urlopen('https://googledictionaryapi.eu-gb.mybluemix.net/?define={}&lang=en'.format(word))
    if defn.getcode() != 200:
        return "[]".encode()
    else:
        return defn.read()

def get_defs(words: List[str]) -> Dict:
    res = {}
    for word in words:
        try:
            res[word] = get_word_def(word)
        except Exception as e:
            with open('./ccl_exceptions.txt', 'a', encoding='utf-8') as f:
                f.write('{}\n{}\n'.format(word, str(e)))
            continue
    return res

def get_words(fin_path: str) -> List:
    res = []
    with open(fin_path, 'r', encoding='utf-8') as f:
        res = re.findall("(?<=English\n).*", ''.join(f.readlines()))
    with open('./ccl_words.txt', 'w', encoding='utf-8') as f:
        f.write('\n'.join(res))
    return []

      
def write_to_file(fin_path: str, fout_path: str) -> bool:
    import json
    with open(fin_path, 'r', encoding='utf-8') as f:
        words = f.readlines()
        words = [w.replace('\n', '') for w in words]
    defns = get_defs(words)
    print("Got definitions")
    with open(fout_path, 'ab') as f:

        return True # successful
    return False
